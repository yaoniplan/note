- #### Use "podman"
    - `vim Dockerfile` # Define the container image
      ```
      FROM python:3.12-alpine
      
      RUN apk add --no-cache git ffmpeg gcc musl-dev python3-dev libffi-dev
      
      WORKDIR /app
      
      # Clone + Install yt-dlp (Do this only once, utilizing caching)
      RUN git clone https://github.com/4ft35t/yt-dlp.git && \
          cd yt-dlp && \
          git checkout fix-ximalaya-album && \
          pip install --no-cache-dir -e . --break-system-packages
      
      WORKDIR /workspace
      ```
    - `vim Makefile` # Automate build and run the container
      ```
      IMAGE_NAME := yt-dlp:custom
      PROXY      := http://host.docker.internal:7890
      
      # Default URL (can be overridden)
      URL ?= https://www.ximalaya.com/album/47769867
      
      # ================= build =================
      build:
      	podman build -t $(IMAGE_NAME) .
      
      # ================= run =================
      run:
      	podman run --rm -it \
      		-v "$(PWD):/workspace:z" \
      		-e http_proxy=$(PROXY) \
      		-e https_proxy=$(PROXY) \
      		-e no_proxy="localhost,127.0.0.1" \
      		$(IMAGE_NAME) \
      		yt-dlp $(URL)
      
      # ================= run with args =================
      # Usage: make dl URL=xxx
      dl:
      	podman run --rm -it \
      		-v "$(PWD):/workspace:z" \
      		-e http_proxy=$(PROXY) \
      		-e https_proxy=$(PROXY) \
      		$(IMAGE_NAME) \
      		yt-dlp $(URL)
      
      # ================= shell =================
      shell:
      	podman run --rm -it \
      		-v "$(PWD):/workspace:z" \
      		-e http_proxy=$(PROXY) \
      		-e https_proxy=$(PROXY) \
      		$(IMAGE_NAME) \
      		sh
      
      # ================= clean =================
      clean:
      	podman rmi $(IMAGE_NAME) || true
      
      .PHONY: build run dl shell clean
      ```
